-- Create users table
CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  fullName VARCHAR(255),
  email VARCHAR(255) UNIQUE,
  phoneNumber VARCHAR(255),
  password VARCHAR(255),
  role VARCHAR(255)
);

-- Create UserDetails table
CREATE TABLE userDetails (
  id INT AUTO_INCREMENT PRIMARY KEY,
  userId INT,
  age INT,
  gender VARCHAR(255),
  weight VARCHAR(10)
);
