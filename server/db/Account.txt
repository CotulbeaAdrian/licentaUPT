-- Create users table
CREATE TABLE users (
  id INT AUTO_INCREMENT PRIMARY KEY,
  username VARCHAR(255) UNIQUE,
  password VARCHAR(255),
  email VARCHAR(255) UNIQUE 
);

-- Create UserDetails table
CREATE TABLE UserDetails (
  uid INT PRIMARY KEY,
  accountId INT,
  age INT,
  fullName TEXT,
  gender TEXT,
  phonenumber TEXT,
  role TEXT,
  weight INT,
  FOREIGN KEY (accountId) REFERENCES Account(id)
);
